Fault-Driven Memory Tracking

Segfaults ain't all bad


                struct Mmap<T: Zeroable> {
                    ptr: *mut u8,
                    len: usize,
                }

                /// This is a trait for stuff in [`Mmap`]
                trait Zeroable {
                    fn zero(&mut self);
                }

                impl Zeroable for Mmap {
                    fn zero(&mut self) {
                        unsafe {
                            libc::memset(self.ptr as *mut c_void, 0, self.len);
                        }
                    }
                }
